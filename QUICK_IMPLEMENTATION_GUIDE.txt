
STEP 4: Fetch real market news (1 hour)
────────────────────────────────────────
```python
from newsapi import NewsApiClient

client = NewsApiClient(api_key="YOUR_KEY")

# Fetch news about stocks
news = client.get_everything(q='Nifty 50', 
                             sort_by='publishedAt',
                             page_size=50)

for article in news['articles']:
    sentiment = finbert(article['title'])
    print(f"{article['title']}: {sentiment}")
```

STEP 5: Analyze sentiment trends (1 hour)
────────────────────────────────────────
```python
import pandas as pd
import numpy as np

sentiments = []
for article in news['articles']:
    s = finbert(article['title'])
    sentiments.append(s['score'] if s['label'] == 'POSITIVE' else -s['score'])

average_sentiment = np.mean(sentiments)
print(f"Market sentiment: {average_sentiment}")
print(f"Direction: {'BULLISH' if average_sentiment > 0 else 'BEARISH'}")
```
